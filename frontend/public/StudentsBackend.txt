📦 Database Schema Suggestions (PostgreSQL)
sql
Copy
Edit
TABLE students (
  id UUID PRIMARY KEY,
  first_name TEXT,
  last_name TEXT,
  gender TEXT,
  date_of_birth DATE,
  class_id UUID REFERENCES classes(id),
  session_id UUID REFERENCES sessions(id),
  guardian_id UUID REFERENCES parents(id),
  photo_url TEXT,
  is_active BOOLEAN DEFAULT TRUE,
  created_at TIMESTAMP,
  updated_at TIMESTAMP
);

TABLE attendance (
  id UUID PRIMARY KEY,
  student_id UUID REFERENCES students(id),
  date DATE,
  status TEXT,  -- Present, Absent, Late
  remarks TEXT
);

TABLE grades (
  id UUID PRIMARY KEY,
  student_id UUID REFERENCES students(id),
  subject_id UUID REFERENCES subjects(id),
  score DECIMAL,
  term TEXT,
  session_id UUID REFERENCES sessions(id),
  level TEXT, -- e.g., A, P, AP
  comment TEXT
);

TABLE parents (
  id UUID PRIMARY KEY,
  full_name TEXT,
  email TEXT,
  phone TEXT,
  relationship TEXT
);
